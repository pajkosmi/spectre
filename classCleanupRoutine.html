Detailed Description

template<typename F>
class CleanupRoutine< F >

An object that calls a functor when it goes out of scope.

int x = 1;
try {
const CleanupRoutine cleanup = [&x]() { x = 3; };
x = 2;
if (do_throw) {
throw std::exception{};
}
} catch (const std::exception&) {
CHECK(x == 3);
}
CHECK(x == 3);
