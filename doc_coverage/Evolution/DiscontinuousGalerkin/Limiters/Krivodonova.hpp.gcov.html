      56             :  * \brief An implementation of the Krivodonova limiter.
      57             :  *
      58             :  * The limiter is described in \cite Krivodonova2007. The Krivodonova limiter
      59             :  * works by limiting the highest derivatives/modal coefficients using an
      60             :  * aggressive minmod approach, decreasing in derivative/modal coefficient order
      61             :  * until no more limiting is necessary. In 3d, the function being limited is
      62             :  * expanded as:
      63             :  *
      64             :  * \f{align}{
      65             :  * u^{l,m,n}=\sum_{i,j,k=0,0,0}^{N_i,N_j,N_k}c^{l,m,n}_{i,j,k}
      66             :  *  P_{i}(\xi)P_{j}(\eta)P_{k}(\zeta)
      67             :  * \f}
      68             :  *
      69             :  * where \f$\left\{\xi, \eta, \zeta\right\}\f$ are the logical coordinates,
      70             :  * \f$P_{i}\f$ are the Legendre polynomials, the superscript \f$\{l,m,n\}\f$
      71             :  * represents the element indexed by \f$l,m,n\f$, and \f$N_i,N_j\f$ and
      72             :  * \f$N_k\f$ are the number of collocation points minus one in the
      73             :  * \f$\xi,\eta,\f$ and \f$\zeta\f$ direction, respectively. The coefficients are
      74             :  * limited according to:
      75             :  *
      76             :  * \f{align}{
      77             :  * \tilde{c}^{l,m,n}_{i,j,k}=\mathrm{minmod}
      78             :  *   &\left(c_{i,j,k}^{l,m,n},
      79             :  *          \alpha_i\left(c^{l+1,m,n}_{i-1,j,k}-c^{l,m,n}_{i-1,j,k}\right),
      80             :  *          \alpha_i\left(c^{l,m,n}_{i-1,j,k}-c^{l-1,m,n}_{i-1,j,k}\right),
      81             :  *     \right.\notag \\
      82             :  *   &\;\;\;\;
      83             :  *          \alpha_j\left(c^{l,m+1,n}_{i,j-1,k}-c^{l,m,n}_{i,j-1,k}\right),
      84             :  *          \alpha_j\left(c^{l,m,n}_{i,j-1,k}-c^{l,m-1,n}_{i,j-1,k}\right),
      85             :  *     \notag \\
      86             :  *   &\;\;\;\;\left.
      87             :  *          \alpha_k\left(c^{l,m,n+1}_{i,j,k-1}-c^{l,m,n}_{i,j,k-1}\right),
      88             :  *          \alpha_k\left(c^{l,m,n}_{i,j,k-1}-c^{l,m,n-1}_{i,j,k-1}\right)
      89             :  *     \right),
      90             :  * \label{eq:krivodonova 3d minmod}

      96             :  *  \mathrm{minmod}(a,b,c,\ldots)=
      97             :  *  \left\{
      98             :  *  \begin{array}{ll}
      99             :  *    \mathrm{sgn}(a)\min(\lvert a\rvert, \lvert b\rvert,
     100             :  *    \lvert c\rvert, \ldots) & \mathrm{if} \;
     101             :  *    \mathrm{sgn}(a)=\mathrm{sgn}(b)=\mathrm{sgn}(c)=\mathrm{sgn}(\ldots) \\
     102             :  *    0 & \mathrm{otherwise}
     103             :  *  \end{array}\right.
